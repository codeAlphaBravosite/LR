            // --- Thumbnail Generation ---
            async function generateThumbnail(videoElement, time) {
                return new Promise((resolve) => {
                    if (!videoElement.src || videoElement.readyState < 2) { 
                        resolve(null); 
                        return;
                    }
                    const wasPaused = videoElement.paused;
                    const originalTime = videoElement.currentTime;
                    videoElement.currentTime = time;
                    const onSeeked = () => {
                        videoElement.removeEventListener('seeked', onSeeked);
                        const canvas = document.createElement('canvas');
                        canvas.width = 160; 
                        canvas.height = 90; 
                        const ctx = canvas.getContext('2d');
                        const videoAspect = videoElement.videoWidth / videoElement.videoHeight;
                        const canvasAspect = canvas.width / canvas.height;
                        let drawWidth, drawHeight, offsetX, offsetY;
                        if (videoAspect > canvasAspect) { 
                            drawHeight = canvas.height;
                            drawWidth = drawHeight * videoAspect;
                            offsetX = (canvas.width - drawWidth) / 2;
                            offsetY = 0;
                        } else { 
                            drawWidth = canvas.width;
                            drawHeight = drawWidth / videoAspect;
                            offsetY = (canvas.height - drawHeight) / 2;
                            offsetX = 0;
                        }
                        ctx.drawImage(videoElement, offsetX, offsetY, drawWidth, drawHeight);
                        if (wasPaused) videoElement.pause(); else videoElement.play();
                        videoElement.currentTime = originalTime;
                        resolve(canvas.toDataURL('image/jpeg', 0.7)); 
                    };
                    videoElement.addEventListener('seeked', onSeeked, { once: true });
                    if (videoElement.currentTime === time && videoElement.seekable.length > 0) {
                        setTimeout(onSeeked, 50);
                    }
                });
            }

                        try {
                            const importedData = JSON.parse(e.target.result);
                            if (importedData && Array.isArray(importedData.projects)) {

            async function selectVideoS2(videoId) {
                currentVideoId_S2 = videoId;
                const project = getCurrentProjectS2();
                const videoData = project.videos.find(v => v.id === videoId);

                if (!videoData) {
                    selectedVideoArea.classList.add('hidden');
                    return;

                document.querySelectorAll('#video-selector-items-s2 .video-select-item').forEach(el => {
                    el.classList.toggle('active', el.dataset.videoId === videoId);
                });

                selectedVideoFilenameS2.textContent = videoData.name;
                selectedVideoArea.classList.remove('hidden');
                
                if (videoFileObjects[videoId]) {
                    screen2VideoPlayer.src = URL.createObjectURL(videoFileObjects[videoId]);

                    const reselectedFile = await promptForVideoFile(videoData.name);
                    if (reselectedFile) {
                        videoFileObjects[videoId] = reselectedFile;
                        screen2VideoPlayer.src = URL.createObjectURL(reselectedFile);
                    } else {
                        screen2VideoPlayer.src = ''; 
                        alert(`Video file for "${videoData.name}" not available. Functions may be limited.`);
                    }
                }
                renderLogsListS2(videoData.logs || []);
            }
            
            function renderLogsListS2(logs) {

                    screen2LogsListItems.classList.add('hidden');
                    screen2LogsEmptyMessage.classList.remove('hidden');

                    screen2LogsListItems.classList.remove('hidden');

                        const item = document.createElement('div');
                        item.className = 'log-item';

                            <div class="details">
                                <p><strong>#${index + 1}: ${log.notes.substring(0, 50)}${log.notes.length > 50 ? '...' : ''}</strong></p>
                                <p class="text-sm text-muted">Time: ${formatTime(log.markIn)} - ${formatTime(log.markOut)} | Rating: ${ratingHtml}</p>
                                ${log.keywords && log.keywords.length > 0 ? `<p class="text-sm text-muted">Keywords: ${log.keywords.join(', ')}</p>` : ''}

            mainVideoPlayerS3.onloadedmetadata = () => {

                     markOutTimecodeS3.textContent = formatTime(s3_markOutTime, true);

                 if (s3_markOutTime < s3_markInTime) { 
                    s3_markInTime = s3_markOutTime; 
                    markInTimecodeS3.textContent = formatTime(s3_markInTime, true);

                markOutTimecodeS3.textContent = formatTime(s3_markOutTime, true);

                star.onclick = () => {
                    s3_currentRating = parseInt(star.dataset.value);

                 star.onmouseover = () => {

            function updateStarDisplayS3() {
                ratingStarsS3.querySelectorAll('.star').forEach((star, index) => {
                    star.innerHTML = `<i class="${index < s3_currentRating ? 'fas' : 'far'} fa-star"></i>`;
                    star.classList.toggle('selected', index < s3_currentRating);
                });
            }
            
            function resetLogFormS3() {
                logNotesS3.value = '';
                logKeywordsS3.value = '';
                s3_currentRating = 0;
                updateStarDisplayS3();
                s3_markInTime = 0;
                s3_markOutTime = mainVideoPlayerS3.duration || 0; 
                markInTimecodeS3.textContent = formatTime(s3_markInTime, true);
                markOutTimecodeS3.textContent = formatTime(s3_markOutTime, true);
                s3_editingLogId = null;
                btnSaveLogS3.innerHTML = '<i class="fas fa-save"></i> Save Log';
                timelineSliderS3.value = 0;
                if(mainVideoPlayerS3.src) mainVideoPlayerS3.currentTime = 0;
            }
            btnClearLogFormS3.onclick = resetLogFormS3;

            btnSaveLogS3.onclick = async () => {
                if (!currentVideoData_S3 || !mainVideoPlayerS3.src) {
                    alert("No video loaded to save a log for.");
                    return;
                }
                if (s3_markOutTime <= s3_markInTime) {
                    if (s3_markInTime === 0 && s3_markOutTime === 0 && mainVideoPlayerS3.duration > 0) {
                         s3_markOutTime = mainVideoPlayerS3.duration;
                    } else if (s3_markOutTime <= s3_markInTime) {
                        alert('Mark Out time must be after Mark In time.');
                        return;
                    }
                }
                const notes = logNotesS3.value.trim();
                const keywords = logKeywordsS3.value.split(',').map(k => k.trim()).filter(k => k);
                if (!notes && s3_currentRating === 0 && keywords.length === 0) {
                    alert("Please provide notes, rating, or keywords for the log.");
                    return;
                }
                mainVideoPlayerS3.pause(); 
                const thumbnail = await generateThumbnail(mainVideoPlayerS3, s3_markInTime);
                const logEntry = {
                    id: s3_editingLogId || generateId(),
                    markIn: s3_markInTime, markOut: s3_markOutTime,
                    notes: notes, rating: s3_currentRating, keywords: keywords, thumbnail: thumbnail
                };
                const project = getCurrentProjectS2(); 
                const videoInProject = project.videos.find(v => v.id === currentVideoData_S3.id);
                if (s3_editingLogId) { 
                    const logIndex = videoInProject.logs.findIndex(l => l.id === s3_editingLogId);
                    if (logIndex > -1) videoInProject.logs[logIndex] = logEntry;
                } else { 
                    if(!videoInProject.logs) videoInProject.logs = []; 
                    videoInProject.logs.push(logEntry);
                }
                videoInProject.logs.sort((a,b) => a.markIn - b.markIn);
                currentVideoData_S3.logs = videoInProject.logs; 
                saveData();
                renderSavedLogsS3();
                resetLogFormS3();
            };

            function renderSavedLogsS3() {
                savedLogsListItemsS3.innerHTML = '';
                if (!currentVideoData_S3 || !currentVideoData_S3.logs || currentVideoData_S3.logs.length === 0) {
                    savedLogsListItemsS3.classList.add('hidden');
                    savedLogsEmptyMessageS3.classList.remove('hidden');
                } else {
                    savedLogsListItemsS3.classList.remove('hidden');
                    savedLogsEmptyMessageS3.classList.add('hidden');
                    currentVideoData_S3.logs.forEach(log => {
                        const item = document.createElement('div');
                        item.className = 'log-item';
                        const ratingHtml = Array(5).fill(0).map((_, i) => 
                            `<i class="${i < log.rating ? 'fas' : 'far'} fa-star" style="color: #f39c12;"></i>`
                        ).join('');
                        item.innerHTML = `
                            ${log.thumbnail ? `<img src="${log.thumbnail}" class="thumbnail" alt="Log thumbnail">` : '<div class="thumbnail">No Preview</div>'}
                            <div class="details">
                                <p><strong>${log.notes.substring(0, 50)}${log.notes.length > 50 ? '...' : ''}</strong></p>
                                <p class="text-sm text-muted">Time: ${formatTime(log.markIn, true)} - ${formatTime(log.markOut, true)}</p>
                                <p class="text-sm text-muted">Rating: ${ratingHtml}</p>
                                ${log.keywords && log.keywords.length > 0 ? `<p class="text-sm text-muted">Keywords: ${log.keywords.join(', ')}</p>` : ''}
                            </div>
                            <div class="actions">
                                <button class="btn-goto-log-s3 secondary" data-log-id="${log.id}"><i class="fas fa-search-location"></i> Go To</button>
                                <button class="btn-edit-log-s3 secondary" data-log-id="${log.id}"><i class="fas fa-edit"></i> Edit</button>
                                <button class="btn-delete-log-s3 danger" data-log-id="${log.id}"><i class="fas fa-trash"></i> Delete</button>
                            </div>
                        `;
                        savedLogsListItemsS3.appendChild(item);
                    });
                }

                document.querySelectorAll('.btn-goto-log-s3').forEach(btn => {
                    btn.onclick = () => {
                        const log = currentVideoData_S3.logs.find(l => l.id === btn.dataset.logId);
                        if (log && mainVideoPlayerS3.src) {
                            mainVideoPlayerS3.currentTime = log.markIn;
                            mainVideoPlayerS3.play();
                            const onTimeUpdateGoTo = () => {
                                if (mainVideoPlayerS3.currentTime >= log.markOut) {
                                    mainVideoPlayerS3.pause();
                                    mainVideoPlayerS3.removeEventListener('timeupdate', onTimeUpdateGoTo);
                                }
                            };
                            if (log.markOut > log.markIn) { 
                                mainVideoPlayerS3.addEventListener('timeupdate', onTimeUpdateGoTo);
                            }
                        }
                    };
                });
                document.querySelectorAll('.btn-edit-log-s3').forEach(btn => {
                    btn.onclick = () => {
                        const log = currentVideoData_S3.logs.find(l => l.id === btn.dataset.logId);
                        if (log) {
                            s3_editingLogId = log.id;
                            logNotesS3.value = log.notes;
                            logKeywordsS3.value = log.keywords.join(', ');
                            s3_currentRating = log.rating;
                            updateStarDisplayS3();
                            s3_markInTime = log.markIn;
                            s3_markOutTime = log.markOut;
                            markInTimecodeS3.textContent = formatTime(s3_markInTime, true);
                            markOutTimecodeS3.textContent = formatTime(s3_markOutTime, true);
                            btnSaveLogS3.innerHTML = '<i class="fas fa-save"></i> Update Log';
                            mainVideoPlayerS3.currentTime = log.markIn;
                            logNotesS3.focus();
                        }
                    };
                });
                document.querySelectorAll('.btn-delete-log-s3').forEach(btn => {
                    btn.onclick = () => {
                        const logId = btn.dataset.logId;
                        const log = currentVideoData_S3.logs.find(l => l.id === logId);
                        showConfirmation(`Are you sure you want to delete this log? "${log.notes.substring(0,20)}..."`, () => {
                            const project = getCurrentProjectS2();
                            const videoInProject = project.videos.find(v => v.id === currentVideoData_S3.id);
                            videoInProject.logs = videoInProject.logs.filter(l => l.id !== logId);
                            currentVideoData_S3.logs = videoInProject.logs; 
                            saveData();
                            renderSavedLogsS3();
                        });
                    };
                });
            }

            // --- Initial Load ---
            loadData();
            renderProjectsScreen(); // This will show the projects screen and handle its empty state.

        }); // End DOMContentLoaded
